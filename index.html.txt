    <script>
        const tg = window.Telegram.WebApp;
        tg.expand(); // –†–∞—Å–∫—Ä—ã—Ç—å –Ω–∞ –≤–µ—Å—å —ç–∫—Ä–∞–Ω

        // –ù–∞—á–∞–ª—å–Ω—ã–µ —Å—Ç–∞—Ç—ã
        let stats = {
            hunger: 100,
            happy: 100,
            energy: 100
        };

        const pet = document.getElementById('pet');

        function updateUI() {
            document.getElementById('hunger-bar').style.width = stats.hunger + '%';
            document.getElementById('hunger-val').innerText = Math.round(stats.hunger) + '%';
            
            document.getElementById('happy-bar').style.width = stats.happy + '%';
            document.getElementById('happy-val').innerText = Math.round(stats.happy) + '%';
            
            document.getElementById('energy-bar').style.width = stats.energy + '%';
            document.getElementById('energy-val').innerText = Math.round(stats.energy) + '%';

            // –ï—Å–ª–∏ —Å—Ç–∞—Ç—ã –Ω–∞ –Ω—É–ª–µ - –ø–∏—Ç–æ–º–µ—Ü –≥—Ä—É—Å—Ç–∏—Ç
            if (stats.hunger <= 0 || stats.happy <= 0 || stats.energy <= 0) {
                pet.innerText = 'üíÄ';
            } else if (stats.hunger < 30 || stats.happy < 30) {
                pet.innerText = 'üòø';
            } else {
                pet.innerText = 'üê±';
            }
        }

        function action(type) {
            pet.classList.remove('bounce');
            void pet.offsetWidth; // –¢—Ä–∏–≥–≥–µ—Ä —Ä–µ—Ñ–ª–æ—É –¥–ª—è –ø–µ—Ä–µ–∑–∞–ø—É—Å–∫–∞ –∞–Ω–∏–º–∞—Ü–∏–∏
            pet.classList.add('bounce');
            
            // –í–∏–±—Ä–∞—Ü–∏—è —Ç–µ–ª–µ—Ñ–æ–Ω–∞ (—Ä–∞–±–æ—Ç–∞–µ—Ç —Ç–æ–ª—å–∫–æ –≤ Telegram App)
            tg.HapticFeedback.impactOccurred('medium');

            if (type === 'feed') {
                stats.hunger = Math.min(100, stats.hunger + 20);
                stats.energy = Math.max(0, stats.energy - 5);
            } else if (type === 'play') {
                stats.happy = Math.min(100, stats.happy + 20);
                stats.energy = Math.max(0, stats.energy - 15);
                stats.hunger = Math.max(0, stats.hunger - 10);
            } else if (type === 'sleep') {
                stats.energy = Math.min(100, stats.energy + 30);
                stats.hunger = Math.max(0, stats.hunger - 10);
            }
            updateUI();
        }

        // –ò–≥—Ä–æ–≤–æ–π —Ü–∏–∫–ª: —Å—Ç–∞—Ç—ã –ø–∞–¥–∞—é—Ç –∫–∞–∂–¥—É—é —Å–µ–∫—É–Ω–¥—É
        setInterval(() => {
            if (stats.hunger > 0) stats.hunger -= 0.5;
            if (stats.happy > 0) stats.happy -= 1;
            if (stats.energy > 0) stats.energy -= 0.2;
            updateUI();
        }, 1000);

        updateUI();
    </script>
